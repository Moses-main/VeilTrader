### 2. Agent Contract
```solidity
// contracts/VeilTrader.sol
contract VeilTrader is IERC8004 {
    struct AgentIdentity {
        address agentAddress;
        string metadataUri;
        bytes32 domain;
        uint256 reputation;
    }
    
    mapping(address => AgentIdentity) public identities;
    
    function registerAgent(string memory uri, bytes32 domain) external {
        identities[msg.sender] = AgentIdentity({
            agentAddress: msg.sender,
            metadataUri: uri,
            domain: domain,
            reputation: 0
        });
    }
}
```

### 3. Trade Receipts

struct TradeReceipt {

    string tokenOut;
